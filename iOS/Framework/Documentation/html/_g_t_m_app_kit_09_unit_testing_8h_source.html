54 // Fails when the |a1|'s drawing in an area |a2| does not equal the image file named |a3|.
55 // See the description of the -gtm_pathForImageNamed method
56 // to understand how |a3| is found and written out.
57 // See the description of the GTMUnitTestView for a better idea
58 // how the view works.
59 // Implemented as a macro to match the rest of the SenTest macros.
60 //
61 // Args:
62 // a1: The object that implements the GTMUnitTestViewDrawer protocol
63 // that is doing the drawing.
64 // a2: The size of the drawing
65 // a3: The name of the image file to check against.
66 // Do not include the extension
67 // a4: contextInfo to pass to drawer
68 // description: A format string as in the printf() function.
69 // Can be nil or an empty string but must be present.
70 // ...: A variable number of arguments to the format string. Can be absent.
71 //

73 #define GTMAssertDrawingEqualToImageNamed(a1, a2, a3, a4, description, ...) \
74  do { \
75  id<GTMUnitTestViewDrawer> a1Drawer = (a1); \
76  NSSize a2Size = (a2); \
77  NSString* a3String = (a3); \
78  void *a4ContextInfo = (a4); \
79  NSRect frame = NSMakeRect(0, 0, a2Size.width, a2Size.height); \

82  } while(0)

105 // A view that allows you to delegate out drawing using the formal
106 // GTMUnitTestViewDelegate protocol
107 // This is useful when writing up unit tests for visual elements.
108 // Your test will often end up looking like this:
109 // - (void)testFoo {
110 // GTMAssertDrawingEqualToFile(self, NSMakeSize(200, 200), @"Foo", nil, nil);
111 // }
112 // and your testSuite will also implement the unitTestViewDrawRect method to do
113 // it's actual drawing. The above creates a view of size 200x200 that draws
114 // it's content using |self|'s unitTestViewDrawRect method and compares it to
115 // the contents of the file Foo.tif to make sure it's valid
116 @interface GTMUnitTestView : NSView {
117  @private
118  id<GTMUnitTestViewDrawer> drawer_; // delegate for doing drawing (STRONG)
119  void* contextInfo_; // info passed in by user for them to use when drawing
120 }
