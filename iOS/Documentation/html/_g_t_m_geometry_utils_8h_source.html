170 //
171 // Returns:
172 // Distance
173 CG_INLINE CGFloat GTMCGDistanceBetweenPoints(CGPoint pt1, CGPoint pt2) {
174  CGFloat dX = pt1.x - pt2.x;
175  CGFloat dY = pt1.y - pt2.y;
176 #if CGFLOAT_IS_DOUBLE
177  return sqrt(dX * dX + dY * dY);
178 #else
179  return sqrtf(dX * dX + dY * dY);
180 #endif
181 }
